    <script>
        $(function () {
            var socket = io();

            $("form").submit(function (e) {
                e.preventDefault();
                const message = $("#msg").val().trim();
                if (message) {
                    socket.emit('chat message', message);
                    $("#msg").val("");
                }
                return false;
            });

            socket.on('chat message', function (msg) {
                $("#messages").append($("<p>").text(msg));
                $("#messages").scrollTop($("#messages")[0].scrollHeight);
            });
        });
